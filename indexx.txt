        1. Go to https://docs.google.com/spreadsheets/u/0/create to create a new Sheet.
        2. Go to Extensions > Apps Script.
        3. Delete any code there and paste the following:

           function doPost(e) {
             var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
             var data = JSON.parse(e.postData.contents);
             sheet.appendRow([new Date(), data.name, data.email, data.phone, data.projectType, data.budget, data.message]);
             return ContentService.createTextOutput(JSON.stringify({"result":"success"})).setMimeType(ContentService.MimeType.JSON);
           }

        4. Click "Deploy" > "New deployment".
        5. Select type: "Web app".
        6. Description: "Contact Form".
        7. Execute as: "Me".
        8. Who has access: "Anyone" (Important!).
        9. Click "Deploy".
        10. Copy the "Web app URL" and paste it below into the GOOGLE_SCRIPT_URL variable.
      */
      const GOOGLE_SCRIPT_URL = "YOUR_GOOGLE_APPS_SCRIPT_URL_HERE"; 

      const handleSubmit = async (e) => {
        e.preventDefault();
        setStatus('submitting');
        
        const formData = new FormData(e.target);
        const data = Object.fromEntries(formData.entries());

        // Simulation if user hasn't replaced the URL yet
        if (GOOGLE_SCRIPT_URL === "YOUR_GOOGLE_APPS_SCRIPT_URL_HERE") {
          console.warn("Please configure your Google Apps Script URL in the code.");
          await new Promise(resolve => setTimeout(resolve, 2000));
          setStatus('success');
          e.target.reset();
          return;
        }

        try {
          // Using no-cors mode because Google Apps Script redirects which triggers CORS errors in fetch
          // However, standard POST with text/plain body usually works best for GAS doPost triggers
          await fetch(GOOGLE_SCRIPT_URL, {
            method: "POST",
            body: JSON.stringify(data),
            // Don't set Content-Type to application/json to avoid preflight OPTIONS request
          });
          
          setStatus('success');
          e.target.reset();
        } catch (error) {
          console.error('Error:', error);
          setStatus('error');
        }
